class Solution {
    public int longestOnes(int[] nums, int k) {
        int ans = 0;
        int st = 0, i;
        for (i = 0; i < nums.length; i++) {
            if (nums[i] == 0) {
                while (k == 0) {
                    ans = Math.max(ans, i - st);
                    if (nums[st++] == 0)
                        k++;
                }
                k--;
            }
        }
        return Math.max(ans, i - st);
    }
}
