    public String reverse(String S) {
        Stack<Character>st = new Stack<>();
        for(char ch : S.toCharArray()){
            st.push(ch);
        }
        StringBuilder t = new StringBuilder();
        while(st.size()>0) t.append(st.pop());
        return t.toString();
    }
}
